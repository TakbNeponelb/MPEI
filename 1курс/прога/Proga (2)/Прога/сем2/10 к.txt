void readthisTXT(PElem &a, int argc, char* argv[]) {
	FILE *txt;
	int Inf;
	fopen_s(&txt, argv[1], "rt");
	while (!feof(txt)) {
		fscanf_s(txt, "%d", &Inf);
		pushback(a, Inf);
	}
	fclose(txt);
}

}

}
