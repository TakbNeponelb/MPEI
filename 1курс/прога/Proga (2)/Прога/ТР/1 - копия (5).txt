void AddTXT(List &a, int argc, char* argv[]) {
	FILE* txt;
	TInfo info;
	int ch, i=0;
	txt = fopen(argv[2], "rt");
	if (!IsNotEmpty(a)) {
		FScanTInfo(info, txt);
		CreateList(a, info, argc, argv);
		while (!feof(txt)) {
			FScanTInfo(info, txt);
			PopBack(a, info);
		}
		return;
	}
	printf("В начало - 1\nВ середину - 2\nВ конец - 3");
	scanf("%d", &ch);
	switch (ch)
	{
	case 1:
		FScanTInfo(info, txt);
		PushBack(a, info, argc, argv);
		while (!feof(txt)) {
			FScanTInfo(info, txt);
			MedBack(a, info);
			NextElem(a);
			i++;
		}
		for (int j = 0; j < i; j++) { PreviousElem(a); }
		break;

	case 2:
		while (!feof(txt)) {
			FScanTInfo(info, txt);
			MedBack(a, info);
			NextElem(a);
			i++;
		}
		for (int j = 0; j < i; j++) { PreviousElem(a); }
		break;
	case 3:
		while (!feof(txt)) {
			FScanTInfo(info, txt);
			PopBack(a, info);
			i++;
		}break;
	default:printf("Неправильная команда!");
		break;
	}
}
